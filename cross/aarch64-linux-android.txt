# e.g. $ANDROID_NDK_HOME/toolchains/llvm/prebuilt/linux-x86_64

c_args = ['-fPIC']
cpp_args = ['-fPIC']

cpp_link_args = ['-fPIC', '-static-libstdc++']

[host_machine]
